1. **–°–æ–∑–¥–∞—Ç—å `sector-filter.js`**
   ```javascript
   import { Logger } from '../utils/logger.js';

   // –ö–µ—à –¥–ª—è –º–µ–º–æ–∏–∑–∞—Ü–∏–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏
   const filterCache = new Map();
   const MAX_CACHE_SIZE = 100;

   /**
    * –£–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω–∞—è —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è —Ç–∏–∫–µ—Ç–æ–≤ –ø–æ —Å–µ–∫—Ç–æ—Ä—É
    * @param {Array} tickets - –ú–∞—Å—Å–∏–≤ —Ç–∏–∫–µ—Ç–æ–≤
    * @param {Object} sectorConfig - –ö–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è —Å–µ–∫—Ç–æ—Ä–∞ –∏–∑ SECTORS_CONFIG
    * @returns {Array} –û—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω–Ω—ã–µ —Ç–∏–∫–µ—Ç—ã
    */
   export function filterTicketsBySector(tickets, sectorConfig) {
     if (!Array.isArray(tickets)) {
       Logger.warn('filterTicketsBySector: tickets is not an array', 'sector-filter');
       return [];
     }

     const cacheKey = getCacheKey(tickets, sectorConfig);
     const cached = filterCache.get(cacheKey);

     if (cached !== undefined) {
       Logger.debug(`Filter cache hit for sector ${sectorConfig.id}`, 'sector-filter');
       return cached;
     }

     // –ü–æ–ª—É—á–∏—Ç—å –Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è —Ñ–∏–ª—å—Ç—Ä–∞

     // –§–∏–ª—å—Ç—Ä–æ–≤–∞—Ç—å —Ç–∏–∫–µ—Ç—ã
     const filtered = tickets.filter(ticket =>
       isTicketInSector(ticket, filterValues, sectorConfig)
     );

     // –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –¥–ª—è –¥–∏–∞–≥–Ω–æ—Å—Ç–∏–∫–∏
     Logger.info(`Filtered ${filtered.length}/${tickets.length} tickets for sector ${sectorConfig.id}`, 'sector-filter', {
       sectorId: sectorConfig.id,
       totalTickets: tickets.length,
       filteredTickets: filtered.length,
       filterValues
     });

     // –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –≤ –∫–µ—à
     filterCache.set(cacheKey, filtered);
     cleanFilterCache();

     return filtered;
   }

   /**
    * –ü–æ–ª—É—á–∏—Ç—å –Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è —Ñ–∏–ª—å—Ç—Ä–∞ –¥–ª—è —Å–µ–∫—Ç–æ—Ä–∞
    */
   function getNormalizedFilterValues(sectorConfig) {
     const { filterValue } = sectorConfig;
     const values = Array.isArray(filterValue) ? filterValue : [filterValue];

     return values.map(value => normalizeFilterValue(value));
   }

   /**
    * –ü—Ä–æ–≤–µ—Ä–∏—Ç—å, –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∏—Ç –ª–∏ —Ç–∏–∫–µ—Ç —Å–µ–∫—Ç–æ—Ä—É
    */
   function isTicketInSector(ticket, filterValues, sectorConfig) {
     const ticketValue = getTicketFieldValue(ticket, 'UF_CRM_7_TYPE_PRODUCT');
     const normalizedTicketValue = normalizeFilterValue(ticketValue);

     return filterValues.some(filterValue => normalizedTicketValue === filterValue);
   }

   /**
    * –ü–æ–ª—É—á–∏—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ –ø–æ–ª—è —Ç–∏–∫–µ—Ç–∞ —Å —É—á–µ—Ç–æ–º —Ä–∞–∑–ª–∏—á–Ω—ã—Ö —Ñ–æ—Ä–º–∞—Ç–æ–≤
    */
   function getTicketFieldValue(ticket, fieldName) {
     return ticket[fieldName] ||
            ticket[fieldName.toLowerCase()] ||
            ticket[camelCase(fieldName)] ||
            null;
   }

   /**
    * –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –∑–Ω–∞—á–µ–Ω–∏—è —Ñ–∏–ª—å—Ç—Ä–∞
    */
   function normalizeFilterValue(value) {
     if (value === null || value === undefined) return null;

     // –î–ª—è –æ–±—ä–µ–∫—Ç–æ–≤ —Å –ø–æ–ª–µ–º value
     if (typeof value === 'object' && value.value !== undefined) {
       value = value.value;
     }

     // –ü—Ä–∏–≤–µ–¥–µ–Ω–∏–µ –∫ —Å—Ç—Ä–æ–∫–µ, trim, uppercase, –∑–∞–º–µ–Ω–∞ –∫–∏—Ä–∏–ª–ª–∏—Ü—ã
     const str = String(value).trim().toUpperCase().replace(/–°/g, 'C');

     return str || null;
   }

   /**
    * –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –∫–ª—é—á–∞ –∫–µ—à–∞
    */
   function getCacheKey(tickets, sectorConfig) {
     const ticketIds = tickets.map(t => t.id || t.ID || '').sort().join(',');
     const filterValues = JSON.stringify(sectorConfig.filterValue);
     return `filter:${sectorConfig.id}:${btoa(ticketIds)}:${btoa(filterValues)}`;
   }

   /**
    * –û—á–∏—Å—Ç–∫–∞ –∫–µ—à–∞ –ø—Ä–∏ –ø—Ä–µ–≤—ã—à–µ–Ω–∏–∏ –ª–∏–º–∏—Ç–∞
    */
   function cleanFilterCache() {
     if (filterCache.size > MAX_CACHE_SIZE) {
       const entriesToDelete = Math.floor(MAX_CACHE_SIZE * 0.2);
       const keys = Array.from(filterCache.keys()).slice(0, entriesToDelete);
       keys.forEach(key => filterCache.delete(key));
     }
   }

   /**
    * –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –≤ camelCase
    */
   function camelCase(str) {
     return str.toLowerCase().replace(/_([a-z])/g, (_, letter) => letter.toUpperCase());
   }

   /**
    * –û—á–∏—Å—Ç–∫–∞ –∫–µ—à–∞ —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏–∏
    */
   export function clearSectorFilterCache() {
     filterCache.clear();
     Logger.info('Sector filter cache cleared', 'sector-filter');
   }
   ```

   ```javascript
   export function getSectorFilterDescription(sectorConfig) {
     const values = getSectorFilterValues(sectorConfig);
     const field = sectorConfig.filterField || 'UF_CRM_7_TYPE_PRODUCT';

     if (values.length === 1) {
       return `${field} = '${values[0]}'`;
     } else {
       return `${field} IN (${values.map(v => `'${v}'`).join(', ')})`;
     }
   }
   ```
