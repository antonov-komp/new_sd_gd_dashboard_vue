  /**

    if (this.serviceCache.has(sectorId)) {

    // –ü–æ–∫–∞ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –º–æ–∫–æ–≤—ã–µ –¥–∞–Ω–Ω—ã–µ
    return this._getMockConfig(sectorId);

    const ServiceClass = this._getServiceClass(config);

  static _getServiceClass(config) {

    if (config.id === '1c') {

    }

  static clearCache() {
    this.serviceCache.clear();
  }
}

/**
 * –°–ø–µ—Ü–∏–∞–ª–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–µ—Ä–≤–∏—Å –¥–ª—è —Å–µ–∫—Ç–æ—Ä–∞ 1–°

    this.useOriginalUfNames = true;

    const targetStages = Object.values(this.stages);
