  } catch (err) {

function prepareChartDataWithComparison() {
  if (!snapshots.value.weekStart || !snapshots.value.weekEnd || !comparison.value) {
    return null;
  }

  const { weekStart, weekEnd, current } = snapshots.value;
  const labels = ['–ù–∞—á–∞–ª–æ –Ω–µ–¥–µ–ª–∏', '–ö–æ–Ω–µ—Ü –Ω–µ–¥–µ–ª–∏'];
  
  if (current) {
    labels.push('–¢–µ–∫—É—â–µ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ');
  }

  const datasets = [];
  const stages = ['formed', 'review', 'execution'];
  const stageNames = {
    formed: '–°—Ñ–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–æ –æ–±—Ä–∞—â–µ–Ω–∏–µ',
    review: '–†–∞—Å—Å–º–æ—Ç—Ä–µ–Ω–∏–µ –¢–ó',
    execution: '–ò—Å–ø–æ–ª–Ω–µ–Ω–∏–µ'
  };

  stages.forEach((stageKey) => {
    const data = [
      weekStart?.statistics?.stages?.[stageKey]?.count || 0,
      weekEnd?.statistics?.stages?.[stageKey]?.count || 0
    ];

    if (current) {
      data.push(current?.statistics?.stages?.[stageKey]?.count || 0);
    }

    // –ü–æ–ª—É—á–µ–Ω–∏–µ —Ç—Ä–µ–Ω–¥–æ–≤ –Ω–∞ –æ—Å–Ω–æ–≤–µ –≤—ã–±—Ä–∞–Ω–Ω–æ–≥–æ —Ç–∏–ø–∞ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è
    const trends = ['stable']; // –ù–∞—á–∞–ª–æ –Ω–µ–¥–µ–ª–∏ (–±–∞–∑–æ–≤–∞—è —Ç–æ—á–∫–∞)
    
    if (comparisonType.value === 'weekStartToWeekEnd') {
      trends.push(comparison.value.weekStartToWeekEnd?.stages?.[stageKey]?.trend || 'stable');
      if (current) {
        trends.push(comparison.value.weekStartToCurrent?.stages?.[stageKey]?.trend || 'stable');
      }
    } else if (comparisonType.value === 'weekEndToCurrent' && current) {
      trends.push('stable'); // –ö–æ–Ω–µ—Ü –Ω–µ–¥–µ–ª–∏ (–±–∞–∑–æ–≤–∞—è —Ç–æ—á–∫–∞ –¥–ª—è —ç—Ç–æ–≥–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è)

      trends.push(comparison.value.weekStartToWeekEnd?.stages?.[stageKey]?.trend || 'stable');
      trends.push(comparison.value.weekStartToCurrent?.stages?.[stageKey]?.trend || 'stable');

    const borderColor = trends.map(trend => getTrendColor(trend, 'border'));

    datasets.push({
      label: stageNames[stageKey],
      data,
      backgroundColor,
      borderColor,
      borderWidth: 2,
      pointBackgroundColor,
      pointBorderColor: borderColor,
      pointRadius: 6,
      pointHoverRadius: 8,
      tension: 0.4
    });
  });

  return {
    labels,
    datasets
  };
}

function getStartOfWeek(date) {
  const d = new Date(date);
  const day = d.getDay();
  const diff = d.getDate() - day + (day === 0 ? -6 : 1); // –ü–æ–Ω–µ–¥–µ–ª—å–Ω–∏–∫
  return new Date(d.setDate(diff));
}

function getEndOfWeek(date) {
  const start = getStartOfWeek(date);
  const end = new Date(start);
  end.setDate(end.getDate() + 6); // –í–æ—Å–∫—Ä–µ—Å–µ–Ω—å–µ
  return end;
}
